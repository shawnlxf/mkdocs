    >

    ```
    str = `Hi, I'm Frank`

    hash = {}

    for i from 0 to str.length-1
    	key = str.get(i)
    	value = hash.get(key, 0) + 1
    	hash.set(key, value)
    
    for key,value from hash 
    	print `${key} 出现了 ${value} 次`
    ```

## 2 数据结构

??? question "如何表示两个数据"

    * 如果顺序有意义

        > 坐标：[x, y], [y, x]

    * 如果顺序无意义

        > 血压值：(120, 80), (80, 120)

??? question "如何表示 N 个数据"

    * 如果顺序有意义

        > 数组：[a1, a2, ..., aN]
        >
        > 要提供 add / indexOf / delete 操作

    * 如果顺序无意义

        > 集合：{a1, a2, ..., aN}
        >
        > 要提供 add / has / delete 操作

??? question "如何表示 N 对 N 数据"

    * 比如 学号

??? question "请写一个 min 函数，要求 min(numbers) 能返回数组 numbers 中的最小数字。"

    ??? note "举例"

        想到数组

        ```
        min([1,2,3,4])
        = min( [ 1, min( [2,3,4] ) ] )
        = min( [ 1, min( [2, min([3,4]) ] ) ] )   // 化繁为简，一招鲜吃遍天，想到用递归
        ```

        ```
        min([numbers[0], min(numbers.slice(1))])   // 递进
        minOf2([a, b]) => a < b ? a : b            // 尽头
        ```

    ??? note "实现"

        ``` javascript
        //let minOf2 = ([a, b]) => a < b ? a : b    // 析构赋值

        let min = (numbers) => {
        	if (numbers.length > 2) {
                return min(
                    [numbers[0], min(numbers.slice(1))]
                )
        	} else {
        		//return minOf2(numbers)
                return Math.min.apply(null, numbers)
        	}
        }
        ```

    ??? note "现成 API"

        * JS 内置了 Math.min 

            > Math 只是一个普通对象，惟一的特例

            * Math.min(1,2)
            * Math.min.call(null,1,2)
            * Math.min.apply(null, [1,2])


??? question "选择排序：请写出一个 sort 函数，要求 sort(numbers) 能返回一个把 numbers 从小到大排列的数组（你可以添加多余的帮助函数）"

    选择排序: 每次取最小的数放前面

    ??? note "举例"

        想到数组

        ```
        sort([3,2,1])
        = min([3,2,1]).concat( sort([3,2]) )
        ...
        = min([3,2,1]).concat( min([3,2]).concat( [3] ) )	// 递归 + 其他函数
        ```

    ??? note "minIndex"

        ``` javascript
        // 取巧的方法
        let minIndex = (numbers) =>
            numbers.indexOf(min(numbers))
        ```

        ``` javascript
        // 更好的方法
        let minIndex = (numbers) => {
	        let index = 0
	        for (let i = 1 ; i < numbers.length; i++) {
		        if (numbers[i] < numbers[index]) {
			        index = i
		        }
	        }

	        return index
        }
        ```

    ??? note "递归实现"

        ``` javascript
        let min = (numbers) => {
            if (numbers.length > 2) {
                return min(
                    [numbers[0], min(numbers.slice(1))]
                )
            } else {
                return Math.min.apply(null, numbers)
            }
        }
  
        let sort = (numbers) => {
	        if (numbers.length > 1) {
        	    let minNum = min(numbers)
	            let index = numbers.indexOf(minNum)
	            numbers.splice(index, 1)
		        return [minNum].concat( sort(numbers) )
	        } else {
		        return numbers;
	        }
        }
        ```

        ``` javascript
        // 循环实现
        let sort = (numbers) => {
            for (let i = 0; i < numbers.length - 1; i++) {
                let index = minIndex(numbers.slice(i)) + i
                //console.log(`${index}: ${numbers[index]}`)
                if (index !== i) {
                    swap(numbers, index, i)
                }
            }

            return numbers;
        }

        let swap = (array, i, j) => {
            let temp = array[i]
            array[i] = array[j]
            array[j] = temp
        }

        let minIndex = (numbers) => {
            let index = 0
            for (let i = 1; i < numbers.length; i++) {
                if (numbers[i] < numbers[index]) {
                    index = i
                }
            }

            return index
        }
        ```


* 所有递归都可以改写成循环

??? note "快速排序"

    ??? note "过程"

        1. 在数据集之中，选择一个元素作为"基准"（pivot）。

        2. 所有小于"基准"的元素，都移到"基准"的左边；所有大于"基准"的元素，都移到"基准"的右边。

        3. 对"基准"左边和右边的两个子集，不断重复第一步和第二步，直到所有子集只剩下一个元素为止。

    ??? note "举例"

        ```
        quickSort([3,2,1])
        = [1,  [2],  3]

        quickSort([4,3,2,1])
        = [1,  [2],  4,3]
        = [1,  [2],  [3],4]
        ```

    ??? note "实现"

        ``` javascript
        let quickSort = arr => {
	        if (arr.length <= 1) {return arr;}
	        let pivotIndex = Math.floor(arr.length / 2)
	        let pivot = arr.splice(pivotIndex, 1)[0]
	        let left = []
	        let right = []
    
	        for (let i = 0; i < arr.length; i++) {
	        	if (arr[i] < pivot) {
	        		left.push(arr[i])
	        	} else {
	        		right.push(arr[i])
	        	}
	        }
    
	        return quickSort(left).concat([pivot], quickSort(right))
        }
        ```

??? note "归并排序"

    ??? note "思路"

        1. 左边一半排序，右边一半排序
        2. 左右两边 merge (合并)


    ??? note "实现"

        ``` javascript
        let mergeSort = arr => {
        	let k = arr.length
        	if (k === 1) { return arr }

        	let left = arr.slice(0, Math.floor(k / 2))
        	let right = arr.slice(Math.floor(k / 2))
        	return merge(mergeSort(left), mergeSort(right))
        }

        let merge = (a, b) => {
        	if (a.length === 0) return b
        	if (b.length === 0) return a
        	return a[0] > b[0] ? 
        		[b[0]].concat(merge(a, b.slice(1))) : [a[0]].concat(merge(a.slice(1), b))
        }
        ```

??? note "计数排序"

    ??? note "思路"

        用一个哈希表作记录
        最后把哈希表的 key 全部打出来，假设 N:m，那么 N 需要打印 m 次

    ??? note "实现"

        ``` javascript
        let countSort = arr => {
	        let hashTable = {}, max = 0, result = []
	        for (let i = 0; i < arr.length; i++) {
	        	if (!(arr[i] in hashTable)) {
	        		hashTable[arr[i]] = 1
 	        	} else {
	        		hashTable[arr[i]] += 1
	        	}
	        	if (arr[i] > max) {
	        		max = arr[i]
	        	}
	        }
	        for (let j = 0; j <= max; j++) {
	        	if (j in hashTable) {
	        		for (let i = 0; i < hashTable[j]; i++){
	        			result.push(j)
	        		}
	        	}
	        }

	        return result
        }
        ```

    ??? note "特点"

        * 数据结构不同

            只遍历数组一遍 (不过还要遍历一次 hashTable)
            
            用空间换时间



* 时间复杂度对比

    * 选择排序 O(n^2)
    * 快速排序 O(n * log2 n)
    * 归并排序 O(n * log2 n)
    * 计数排序 O(n + max)
